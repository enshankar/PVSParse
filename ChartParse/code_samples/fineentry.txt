fine_entry?(len, G, (s: strings(len)))
                      (A : scaffold(len),
                        (i : upto(len)), n, (u: ent))
   : bool =
   CASES u OF
      fail(d): good_fail?(len, G, s)(A, i, n, d),
      loop: loop_ready?(len, G, A,  i, n), 
      good(d, sp):  sp <= len - i AND
                    good_good?(len, G, s)(A, i, n, d, sp)
      ELSE fine_push_entry?(len)(u) AND
              push_or_pending?(A(pos(u))(nt(u)))
   ENDCASES
